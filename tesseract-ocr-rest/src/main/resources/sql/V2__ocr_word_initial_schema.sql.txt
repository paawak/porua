CREATE MEMORY TABLE IF NOT EXISTS raw_image ( 
	id INTEGER GENERATED ALWAYS AS IDENTITY(START WITH 1) PRIMARY KEY, 
	name VARCHAR(200) UNIQUE NOT NULL,
	language VARCHAR(25) NOT NULL
);

CREATE MEMORY TABLE IF NOT EXISTS ocr_word ( 
	id INTEGER GENERATED ALWAYS AS IDENTITY(START WITH 1) PRIMARY KEY, 
	page_sequence_id INTEGER NOT NULL,
	raw_text CLOB NOT NULL,
	corrected_text CLOB,
    raw_image_id INTEGER NOT NULL REFERENCES raw_image (id),
    x1 INTEGER NOT NULL, 
    y1 INTEGER NOT NULL, 
    x2 INTEGER NOT NULL,
    y2 INTEGER NOT NULL, 
    confidence  DECIMAL NOT NULL	
);
